\section*{\textcolor{addiuvaBlue}{6. Metodología}}

\subsection*{6.1 Cálculo de SLA (Service Level Agreement)}

\textbf{Definición}: Porcentaje de servicios atendidos dentro del tiempo máximo establecido.

\textbf{Fórmula}:
$$ SLA = \frac{\text{Servicios Cumple}}{\text{Total Servicios Válidos}} \times 100 $$

\textbf{Umbrales por Tipo de Servicio}:
\begin{itemize}
    \item Contactación Vial Local: 45 minutos
    \item Contactación Vial Foráneo: 90 minutos
    \item Contactación Legal Local: 35 minutos
    \item Contactación Legal Foráneo: 60 minutos
    \item Asignación: 10 minutos
\end{itemize}

\textbf{Exclusiones}:
\begin{itemize}
    \item Servicios Programados (\texttt{servicios\_programados = "No"})
    \item Estados: Cancelado, Fallida, Anulado
    \item Keywords en motivos: "Cita", "Agendada", "Programada", "Posterior"
\end{itemize}

\subsection*{6.2 Cálculo de NPS (Net Promoter Score)}

\textbf{Definición}: Índice de lealtad y recomendación del cliente.

\textbf{Fórmula}:
$$ NPS = \%Promotores - \%Detractores $$

\textbf{Clasificación (Escala 0-10)}:
\begin{itemize}
    \item \textbf{Promotores}: Calificación = 10
    \item \textbf{Pasivos}: Calificación 7-9
    \item \textbf{Detractores}: Calificación 0-6
\end{itemize}

\textbf{Clasificación (Escala 1-5)}:
\begin{itemize}
    \item \textbf{Promotores}: Calificación = 5
    \item \textbf{Pasivos}: Calificación = 4
    \item \textbf{Detractores}: Calificación 1-3
\end{itemize}

    \item \textbf{Periodo}: Enero - Octubre 2025
